\section{Petersson 内积}
% ----------------------------------------------------------------------------

\begin{definition}[Petersson 内积]
对 $f, g \in \Sk_k(\Gamma)$（至少一个为尖点形式），定义
\[
  \langle f, g \rangle = \frac{1}{[\SL_2(\Z):\Gamma]}
  \int_{\Gamma \backslash \HH} f(\tau) \overline{g(\tau)}\, (\im \tau)^k
  \, \frac{dx\,dy}{y^2},
\]
其中 $\tau = x + iy$。
\end{definition}
