\section{紧黎曼面 $X(\Gamma)$}
% ----------------------------------------------------------------------------

\begin{definition}
设 $\Gamma$ 为 $\SL_2(\Z)$ 的同余子群，\textbf{模曲线} $X(\Gamma)$ 定义为
紧黎曼面 $\Gamma \backslash \HH^*$。特别地：
\begin{itemize}
  \item $X_0(N) = X(\GammaO(N))$,
  \item $X_1(N) = X(\GammaI(N))$,
  \item $X(N) = X(\Gamma(N))$.
\end{itemize}
\end{definition}
