\section{$\Mk_k(\SL_2(\Z))$ 的维数}
% ----------------------------------------------------------------------------

\begin{theorem}[维数公式]
\label{thm:dim-formula-sl2z}
对非负偶整数 $k$，
\[
  \dim \Mk_k(\SL_2(\Z)) =
  \begin{cases}
    \lfloor k/12 \rfloor     & \text{若 } k \equiv 2 \pmod{12}, \\
    \lfloor k/12 \rfloor + 1 & \text{否则},
  \end{cases}
\]
且 $\dim \Sk_k(\SL_2(\Z)) = \dim \Mk_k(\SL_2(\Z)) - 1$（$k \geq 2$）。
\end{theorem}
