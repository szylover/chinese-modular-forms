\section{$L$-函数的定义}
% ----------------------------------------------------------------------------

\begin{definition}
设 $f(\tau) = \sum_{n=1}^{\infty} a_n q^n \in \Sk_k(\GammaO(N))$ 为规范化新形式，
其 \textbf{$L$-函数}定义为 Dirichlet 级数
\[
  L(f, s) = \sum_{n=1}^{\infty} \frac{a_n}{n^s},
  \quad \Re(s) > \frac{k+1}{2}.
\]
\end{definition}
