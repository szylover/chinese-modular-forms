\begin{definition}[Eisenstein 级数]
对偶整数 $k \geq 4$，\textbf{Eisenstein 级数}定义为
\[
  G_k(\tau) = \sum_{\substack{(c,d) \in \Z^2 \\ (c,d) \neq (0,0)}}
  \frac{1}{(c\tau + d)^k}.
\]
\end{definition}
