\section{Dirichlet 特征}
% ----------------------------------------------------------------------------

\begin{definition}[Dirichlet 特征]
模 $N$ 的 \textbf{Dirichlet 特征}是群同态
$\chi\colon (\Z/N\Z)^\times \to \C^\times$。
将其扩展到 $\Z$ 上：若 $\gcd(n, N) > 1$，令 $\chi(n) = 0$。
\end{definition}
