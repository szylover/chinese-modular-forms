\section{全纯函数与亚纯函数}
% ----------------------------------------------------------------------------

\begin{definition}[全纯函数]
设 $U \subset \C$ 为开集，函数 $f\colon U \to \C$ 在 $z_0 \in U$ 处
\textbf{全纯}（holomorphic），若极限
\[
  f'(z_0) = \lim_{h \to 0} \frac{f(z_0 + h) - f(z_0)}{h}
\]
存在。若 $f$ 在 $U$ 上每一点都全纯，则称 $f$ 在 $U$ 上全纯。
\end{definition}
