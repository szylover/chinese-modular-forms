\section{$\Q$ 上椭圆曲线的 Mordell-Weil 定理}
% ----------------------------------------------------------------------------

\begin{theorem}[Mordell-Weil]
设 $E/\Q$ 为椭圆曲线，则 $E(\Q)$ 是有限生成 Abel 群：
\[
  E(\Q) \cong \Z^r \oplus E(\Q)_{\mathrm{tors}},
\]
其中 $r = \rank(E/\Q) \geq 0$。
\end{theorem}
